// Theme initialization function initTheme() { const savedTheme = localStorage.getItem('rollcall_theme'); const body = document.body; if (savedTheme === 'light') { body.classList.add('light-theme'); body.classList.remove('dark-theme'); } else if (savedTheme === 'dark') { body.classList.add('dark-theme'); body.classList.remove('light-theme'); } else if (savedTheme === 'system') { const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches; if (prefersDark) { body.classList.add('dark-theme'); body.classList.remove('light-theme'); } else { body.classList.add('light-theme'); body.classList.remove('dark-theme'); } } else { // Default to dark body.classList.add('dark-theme'); } } initTheme();
